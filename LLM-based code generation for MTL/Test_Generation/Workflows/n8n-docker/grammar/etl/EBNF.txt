                   | annotationBlock
                   | operationDeclaration
                   | post
                   ;

pre  ::= "pre"  [ NAME ]  statementBlock ;
post ::= "post" [ NAME ]  statementBlock ;

guard ::= "guard"  expressionOrStatementBlock ;

extendz ::= "extends"  NAME  { "," NAME } ;

(* Terminals / tokens *)
NAME ::= Identifier ;   (* defined in the lexer / elsewhere *)
